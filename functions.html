ques 7:- Write a function that accepts parameter n and returns factorial of n
ans:- 
  function factorial(n){
    let answer = 1;
    if (n == 0 || n == 1){
      return answer;
    }else{
      for(var i = n; i >= 1; i--){
        answer = answer * i;
      }
      return answer;
    }  
  }

  let n = 4;
  answer = factorial(n)
  console.log("The factorial of " + n + " is " + answer);
output:- 
  The factorial of 4 is 24
